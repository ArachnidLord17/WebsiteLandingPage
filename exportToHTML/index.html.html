<script>
document.getElementById('contact-form').addEventListener('submit', function(event) {
    event.preventDefault();

    const form = event.target;
    const data = new FormData(form);

    // Set the CC field to the sender's email
    data.set('_cc', form.email.value);

    fetch(form.action, {
        method: 'POST',
        body: data,
        headers: {
            'Accept': 'application/json'
        }
    }).then(response => {
        if (response.ok) {
            document.getElementById('success-message').style.display = 'block';
            document.getElementById('error-message').style.display = 'none';
            form.reset();
            setTimeout(() => {
                document.getElementById('success-message').style.display = 'none';
                location.reload(); // Refresh the page after 5 seconds
            }, 5000); // Hide success message and refresh after 5 seconds
        } else {
            response.json().then(data => {
                if (Object.hasOwn(data, 'errors')) {
                    document.getElementById('error-message').innerText = data["errors"].map(error => error["message"]).join(", ");
                } else {
                    document.getElementById('error-message').innerText = 'Oops! There was a problem submitting your form';
                }
                document.getElementById('success-message').style.display = 'none';
                document.getElementById('error-message').style.display = 'block';
            });
        }
    }).catch(error => {
        document.getElementById('success-message').style.display = 'none';
        document.getElementById('error-message').innerText = 'Oops! There was a problem submitting your form';
        document.getElementById('error-message').style.display = 'block';
    });
});
</script>
